83def Tracking(fileinput, fileoutput) :
84 """!Docs for Tracking
85 This function performs the second abstractin of data by calling the Retina, RectifyTracks and UnifyTree macros.
86 It subdivides the data into a number of bunches equal to the number of pc cores, then each cores performs data abstraction on different bunches.
87 At the end the UnifyTree macro is called and all created tree are merge into a unique tree put into .root file.
88 It is advised to not interact with the pc throughout this process as this may cause slow-downs due to deprivation of resources.
89 @param fileinput Name of the input file with extension, the path is not necessary.
90
91 @param fileoutput Name of the outputfile with no extension, the output file is put in the folder Dataroot/TrackingAbstractionData with the name fileoutput.root
92 """
93 ROOT.gROOT.LoadMacro(os.path.join("Core", "TrackingRetina", "Retina.cpp++O"))
94 ROOT.gROOT.LoadMacro(os.path.join("Core", "TrackingRetina", "UnifyTree.cpp"))
95 ROOT.gROOT.LoadMacro(os.path.join("Core", "TrackingRetina", "RectifyTracks.cpp"))
96 def AnalysePartOfData(j):
97 logger.info("Inizio processo numero " + str(j))
98 Stringsave = os.path.join("Data", "Dataroot","TrackingAbstractionData", fileoutput +str(j)+ ".root")
99 ROOT.Retina(fileinput, Stringsave, j, os.cpu_count())
100 logger.info("Fine processo numero " + str(j))
101
102 if os.cpu_count()<2:
103 logger.warning("Attenzione: Il pc che stai utilizzando sembrerebbe avere meno di due core disponibili, potrebbe volerci un pÃ² di tempo... Sui dataset forniti fino ad un massimo di 2 ore nel caso peggiore.")
104
105 Iterations = range(os.cpu_count())
106 pool = mp.Pool(processes = os.cpu_count())
107 results = pool.map(AnalysePartOfData, Iterations)
108 pool.close() #boh non la chiude da sola! bug DI PYTHON!!!!!!
109 UtilsList = [None]*os.cpu_count()

113 UtilsList[i] = os.path.join("Data" , "Dataroot" ,"TrackingAbstractionData",fileoutput +str(item) +".root")

116 for i, item in enumerate(UtilsList):
